        function addFilaTarea() {
            const tbody = document.querySelector('#tabla-tareas tbody');
            const row = tbody.insertRow();
            row.className = "fila-tarea";
            row.innerHTML = `
                <td><input type="time" class="t-parada" style="width:100%"></td>
                <td><input type="time" class="t-arranque" style="width:100%"></td>
                <td><select class="t-tarea" style="width:100%">${tareas.map(t => `<option>${t}</option>`).join('')}</select></td>
                <td><select class="t-activo" style="width:100%">${activos.map(a => `<option>${a}</option>`).join('')}</select></td>
                <td><input type="text" class="t-obs" style="width:100%"></td>`;
        }

        async function enviarDatosASheets() {
            const btn = document.getElementById('btnEnviar');
            btn.innerText = "Enviando...";
            btn.disabled = true;

            const tareasArray = [];
            document.querySelectorAll('.fila-tarea').forEach(fila => {
                const p = fila.querySelector('.t-parada').value;
                const a = fila.querySelector('.t-arranque').value;
                if(p || a || fila.querySelector('.t-obs').value) {
                    tareasArray.push({
                        p: p,
                        a: a,
                        t: fila.querySelector('.t-tarea').value,
                        ac: fila.querySelector('.t-activo').value,
                        o: fila.querySelector('.t-obs').value
                    });
                }
            });

            if (tareasArray.length === 0) tareasArray.push({p:'', a:'', t:'Sin tareas', ac:'', o:''});

            const data = {
                fecha: document.getElementById('f_fecha').value,
                turno: document.getElementById('f_turno').value,
                operario: document.getElementById('f_operario').value,
                lote: document.getElementById('f_lote').value,
                h_mol_ini: document.getElementById('h_mol_ini').value,
                h_prod_ini: document.getElementById('h_prod_ini').value,
                z1_ini: document.getElementById('z1_ini').value,
                z2_ini: document.getElementById('z2_ini').value,
                h_mol_fin: document.getElementById('h_mol_fin').value,
                h_prod_fin: document.getElementById('h_prod_fin').value,
                z1_fin: document.getElementById('z1_fin').value,
                z2_fin: document.getElementById('z2_fin').value,
                tareas: tareasArray
            };

            try {
                await fetch(SCRIPT_URL, {
                    method: 'POST',
                    mode: 'no-cors',
                    body: JSON.stringify(data)
                });
                alert("Parte guardado correctamente.");
                window.location.reload();
            } catch (e) {
                alert("Error de conexión.");
                btn.disabled = false;
                btn.innerText = "REINTENTAR ENVÍO";
            }
        }

        // --- GESTIÓN MODAL ---
        function openModal(type) {
            currentEditType = type;
            document.getElementById('editModal').style.display = "block";
            document.getElementById('modalTitle').innerText = "Editar " + type.toUpperCase();
            updateModalList();
        }
        function updateModalList() {
            const listDiv = document.getElementById('modalList');
            let items = (currentEditType === 'operarios') ? operarios : (currentEditType === 'activos') ? activos : tareas;
            listDiv.innerHTML = items.map((item, index) => `
                <div class="list-item">
                    <span>${item}</span>
                    <button style="color:red; border:none; background:none; cursor:pointer;" onclick="removeItem(${index})">✕</button>
                </div>`).join('');
        }
        function saveNewItem() {
            const val = document.getElementById('newItemInput').value.trim();
            if(!val) return;
            if (currentEditType === 'operarios') operarios.push(val);
            else if (currentEditType === 'activos') activos.push(val);
            else tareas.push(val);
            document.getElementById('newItemInput').value = "";
            updateModalList();
        }
        function removeItem(index) {
            let list = (currentEditType === 'operarios') ? operarios : (currentEditType === 'activos') ? activos : tareas;
            list.splice(index, 1);
            updateModalList();
        }
        function closeModal() { document.getElementById('editModal').style.display = "none"; }

        function exportToExcel() {
            const d1 = document.getElementById('exp_desde').value;
            const d2 = document.getElementById('exp_hasta').value;
            if(!d1 || !d2) { alert("Selecciona primero las fechas"); return; }
            
            const dummyData = [{ Aviso: "Abre tu Google Sheets para descargar el reporte real. Este es un archivo de prueba." }];
            const ws = XLSX.utils.json_to_sheet(dummyData);
            const wb = XLSX.utils.book_new();
            XLSX.utils.book_append_sheet(wb, ws, "Produccion");
            XLSX.writeFile(wb, `Tradebe_Reporte_${d1}_a_${d2}.xlsx`);
        }
